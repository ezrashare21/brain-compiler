import sys
pointer = 0
table = []
for x in range(50):
  table.append(0)
def debug():
  print("tbl: " + str(table))
  print("pointer: " + str(pointer))
def left():
  global pointer
  pointer += 1
  if pointer > (len(table) - 1):
    pointer = 1
def right():
  global pointer
  pointer -= 1
  if pointer < 0:
    pointer = len(table)
def up():
  global pointer
  table[pointer] += 1
  if table[pointer] > 255:
    table[pointer] = 0
def down():
  global pointer
  table[pointer] -= 1
  if table[pointer] < 0:
    table[pointer] = 255
def printb():
  global pointer
  sys.stdout.write(chr(table[pointer]))
def inputb():
  global pointer
  table[pointer] = ord(input("")[0])
